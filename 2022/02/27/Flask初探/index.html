flask 基本配置

from flask import Flask
app = Flask(__name__)  # 开头必写，创建一个Flask对象从而进行后续操作


@app.route('/')
def hello_world():  # 这是视图函数
    # 控制相关的代码写在这里面


if __name__ == '__main__':
    app.run() # 用刚刚创建的Flask对象控制程序运行
